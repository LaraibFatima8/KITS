// Productivity Analytics
void trackProductivity() {
  static unsigned long sessionStart = 0;
  static int pomodoroCount = 0;
  
  uint8_t gesture = apds.readGesture();
  
  if(gesture == APDS9960_UP && sessionStart == 0) {
    sessionStart = millis();
    showOLED("üéØ Focus Started");
    logActivity("FOCUS_START");
  }
  
  if(gesture == APDS9960_DOWN && sessionStart > 0) {
    unsigned long sessionTime = millis() - sessionStart;
    pomodoroCount++;
    showOLED("‚úÖ Session " + String(pomodoroCount));
    saveToCloud(sessionTime, pomodoroCount);
    sessionStart = 0;
  }
}

// Gesture Art Generator
void createArt() {
  uint8_t gesture = apds.readGesture();
  static int x = 64, y = 32; // Center of OLED
  
  display.clearDisplay();
  
  switch(gesture) {
    case APDS9960_UP:
      drawLine(x, y, x, y-10);
      y -= 10;
      break;
    case APDS9960_DOWN:
      drawLine(x, y, x, y+10);
      y += 10;
      break;
    case APDS9960_LEFT:
      drawLine(x, y, x-10, y);
      x -= 10;
      break;
    case APDS9960_RIGHT:
      drawLine(x, y, x+10, y);
      x += 10;
      break;
  }
  
  display.display();
}
